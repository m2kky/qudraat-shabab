# قواعد Firestore آمنة مع نظام المستخدمين
# انسخ هذه القواعد في Firebase Console > Firestore > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // قواعد للمدربين - القراءة للجميع، الكتابة للإدمن فقط
    match /instructors/{instructorId} {
      allow read: if true; // أي شخص يمكنه قراءة المدربين
      allow write: if request.auth != null && 
                      exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
                      get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    
    // قواعد للأحداث - القراءة للجميع، الكتابة للإدمن فقط
    match /events/{eventId} {
      allow read: if true; // أي شخص يمكنه قراءة الأحداث
      allow write: if request.auth != null && 
                      exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
                      get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    
    // قواعد للتسجيلات - القراءة للإدمن، الكتابة للجميع
    match /registrations/{registrationId} {
      allow read: if request.auth != null && 
                     exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
                     get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
      allow create: if true; // أي شخص يمكنه التسجيل
    }
    
    // قواعد للمستخدمين - القراءة والكتابة للمستخدم نفسه فقط
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
}

# هذه القواعد تتطلب:
# 1. إنشاء collection "users" مع role="admin" للإدمن
# 2. تسجيل دخول المستخدمين
# 3. نظام المصادقة
